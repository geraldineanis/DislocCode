15def x_t(t,xi,m, B, F):
16 """
17 Analytic solution to the equation of motion mx" + Bx' = F
18
19 Parameters
20 ----------
21 t : numpy.ndarray
22 Time - times at which position is calculated.
23 xi : float
24 Initial dislocation position.
25 m : float
26 Dislocation effective mass.
27 B : float
28 Drag coefficient.
29 F : float
30 Force on the dislocation.
31 Returns
32 -------
33 float
34 Dislocation position timeseries.
35 """
36 return xi + ((F/B)*(t + (m/B)*(np.exp(-(B/m)*t) - 1.0)))

38def x_derivatives(x,t,c0,c1,c2):
39 """
40 Defines the derivatives needed to solve the equation c0 x" + c1 x' = c2
41 This is the non-dimensional equation of motion
42
43 Parameters
44 ----------
45 x : numpy.ndarray
46 LHS row vector (dependent variable)
47 t : numpy.ndarray
48 time (the independent variable)
49 c0 : float
50 Non-dimensional parameter related to the effective mass.
51 c1 : float
52 Non-dimensional parameter related to the drag coefficient.
53 c2 : float
54 Non-dimensional parameter related to the force.
55
56 Returns
57 -------
58 numpy.ndarray
59 Derrivates.
60 """
61 dxdt = [x[1], np.exp(c0)*(np.exp(c2) - np.exp(c1)*x[1])]
62 return dxdt

64def ODE_solution(t,x0,t_data):
65 """
66 Solves the non-dimensional equation of motion c0 x" + c1 x' = c2
67 Requires derivatives to be defined previously.
68
69 Parameters
70 ----------
71 t : numpy.ndarray
72 Parameter vector
73 x0 : numpy.ndarray
74 Initial position and velocity.
75 t_data : numpy.ndarray
76 Time - times at which equation is solved.
77 Returns
78 -------
79 numpy.ndarray
80 Non-dimensional dislocation position time series.
81 numpy.ndarray
82 Non-dimensional dislocation velocity time series.
83 """
84 c0, c1, c2 = t # Unpack parameters
85 position, velocity = odeint(x_derivatives, x0, t_data, args=(c0,c1,c2)).T
86 return position, velocity

137def c0_to_m(c0,ts,xs,Fs):
138 """
139 Calculates m, the dislocation effective mass, from its corresponding

143 ----------
144 c0 : float
145 ts : float
146 xs : float
147 Fs : float

193def get_posterior(data, n_samples, burn, dims=3):
194 """
195 Calculates parameter posterior distributions from DE-MC accepted samples.
196
197 Parameters
198 ----------
199 data : np.ndarray
200 DE-MC output.
201 n_samples : int
202 Number of DE_MC samples.
203 burn : int
204 Number of accepted samples to discard.
205 dims : int
206 Number of parameters.
207
208 Returns
209 -------
210 numpy.ndarray
211 Marginal parameter posterior distributions.
212 numpy.ndarray
213 Marginal parameter posterior distributions.
214 numpy.ndarray
215 Parameter means.
216 numpy.ndarray
217 Joint posterior distribution covariance matrix.
218 numpy.ndarray
219 Parameter standard deviations.
220 """
221 # All posterior samples
222 posterior = np.array(data["posterior"][burn:n_samples])
223 # Marginal posterior samples
224 c_dist = np.array([np.array(posterior[:,i]) for i in range(dims)])
225 # Parameter means
226 c_means = np.array([c_dist[i].mean() for i in range(dims)])
227 # Posterior covariance matrix
228 post_cov = np.array(np.cov(posterior, rowvar=False))
229 # Parameter standard deviations
230 c_std = np.array([np.sqrt(post_cov[i][i]) for i in range(dims)])
231
232 return c_dist, c_means, post_cov, c_std

234def stat_convergence(c_dist, dims, write_out=True):
235 """
236 Function to calculate the mean and standarde deviation of a parameter dataset
237 with increasing sample size.
238
239 Parameters
240 ----------
241 c_dist : numpy.array
242 Accepted samples for each parameter. This should be an N x M
243 array where N is equal to dims and M is the number of samples.
244 dims : int
245 Number of fitting parameters.
246 write_out : bool
247 If True, writes out the results to stats_convg.txt
248 """
249 means = np.array([[c_dist[i][:N].mean() for N in range(1,len(c_dist[i]))] for i in range(dims)])
250 std_devs = np.array([[c_dist[i][:N].std() for N in range(1,len(c_dist[i]))] for i in range(dims)])
251
252 if write_out:
253 f = open("stats_convg.txt", "w")
254
255 for i in range(len(means)):
256 f.write(f"{means[i]} {std_devs[i]} \n")
257 f.close()
258
259 return means, std_devs
