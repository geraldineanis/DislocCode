188def get_ellipse(cov, mean, color, label):
189 r2 = st.chi2.ppf(2*st.norm.cdf(2) - 1 ,2)
190
191 val, vec = np.linalg.eigh(cov)
192 width, height = 2*np.sqrt(val[:,None]*r2)
193 rotation = np.degrees(np.arctan2(*vec[::-1,0]))
194
195 return Ellipse(xy=mean, width=width, height=height, fill=False, color=color, label=label, zorder=3 ,angle=rotation)
