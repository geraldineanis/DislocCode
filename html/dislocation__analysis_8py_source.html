7"""
8Dislocation trajectory analysis module.
9
10This module contains functions that are used in the script "dislocation_script.py".
11
12The functions contained in this module allow the user to analyse Molecular Dynamics (MD) trajectories of straight edge dislocations
13generated using LAMMPS.
14
15The script "ovito_disloc.py" post-processes dislocation trajectories generated as LAMMPS dumps files containing the unwrapped coordinates of atoms.
16The functions in this script use as input the text files generated by "ovito_disloc.py", the format of which is described in further detail in the
17its documentation.
18
19The function "track_disloc" included here allows the user to track dislocations across the various frames of a trajectory, since in a simulation containing
20multiple dislocations, different dislocations are indexed differently by the DXA tool. Accordingly, this function allows the user to "track" a given
21dislocation throught a trajectory and analyse its dynamics.
22
23The script "dislocation_script.py" shows how the functions included in this module can be used to process dislocation trajectories prior to carrying out
24the dislocation tracking.

27def read_disloc_data(disloc_file):
28 """
29 Read in dislocation position data obtained from
30 processing MD trajectory using OVITO sctript provided.
31
32 Parameters
33 ----------
34 disloc_file : str
35 Dislocation data file name.
36
37 Returns
38 -------
39 numpy.ndarray
40 Dislocation line x, y, and z coordinates.
41 """
42 with open(disloc_file) as f:
43 ndisloc = int(f.readline()) # number of dislocations
44 indices = [] # indices of dislocations
45 lens = [] # lengths of dislocations
46 dislocations = [] # coordinates of dislocations
47
48 # Read dislocation coordinates for each dislocation found
49 for i in range(ndisloc):
50 disloc_i = []
51 index = indices.append(int(f.readline().strip()))
52 length = int(f.readline().strip())
53 lens.append(length)
54 for i in range(length):
55 coords = f.readline().strip()
56 coords = np.array(coords.split())

58 disloc_i.append(coords)
59 dislocations.append(disloc_i)

87def wrap_coords(dislocation, ax, extent):
88 """
89 Function to wrap the coordinates of a dislocation line along a given axis.
90
91 Parameters
92 ----------
93 dislocation : numpy.ndarray
94 Dislocation line coordinates.
95 ax : str
96 "x", "y", or "z" - direction for wrapping.
97 extent : float
98 Cell length (in Angstroms) in direction of wrapping.
99
100 Returns
101 -------
102 numpy.ndarray
103 Modified dislocation coordinates after wrapping.
104 """
105 if ax == "x":
106 j = 0
107 elif ax == "y":
108 j = 1
109 elif ax == "z":
110 j = 2
111 else:
112 print("Error: ax must be set to 'x', 'y', or 'z'")
113 sys.exit()
114
115 for i in range(len(dislocation)):
116 if dislocation[i][j] < 0.0:
117 dislocation[i][j] = dislocation[i][j] + extent
118 if dislocation[i][j] > extent:
119 dislocation[i][j] = dislocation[i][j] - extent
120
121 return dislocation

167 """

170 at every simulation timestep.
171
172 Parameters

178
179 Returns
180 -------
181 numpy.ndarray
182 Perfect dislocation x-coordinates timeseries.

219def track_disloc(avg, x_lim):
220
221 """
222 Track dislocation lines across multiple frames.
223 - Dislocations are tracked by checking the minimum distance moved by a dislocation between frames.
224
225 - Adapted from the approach in:
226 https://pysource.com/2021/10/05/object-tracking-from-scratch-opencv-and-python/
227
228 - Assumptions:
229 1. The dislocation only moves in the x direction
230 2. The number of dislocations is constant across all frames
231 i.e. no dislocations are generated nor lost throughout the analysed trajectory.
232
233 Parameters
234 ----------
235 avg : numpy.ndarray
236 Average x, y, and z coordinates of dislocation lines in the analysed trajectory.
237 x_lim : float
238 Simulation cell length in x (Angstroms).
239 n_frames : int
240 Number of frames in trajectory.
241
242 Returns
243 -------
244 list
245 Sorted dislocation x-coordinates across trajectory frames.
246 list
247 Sorted coordinates of dislocation line vertices across frames.
248 list
249 Timesteps.
250
251 """
252 # Tracking ID
253 track_ID = 0
254
255 # Dictionary to store dislocations
256 tracking_objects = defaultdict(list)
257
258 # Compare x coordinate of average position
259 # first we only compare the first and second frames
260 # to determine object id at beginning
261 # This sorts the first two points well
262
263 # Cutoff distance
264 d_cutoff = 5.0 # Angstroms
265
266 for i in range(1,2):
267 for j in range(len(avg[i])):
268 for k in range(len(avg[i])):
269 displacement = avg[i][j][0][0] - avg[i-1][k][0][0]
270 if abs(displacement) < d_cutoff:
271 pbc = False
272 tracking_objects[track_ID].append([i-1,np.array([avg[i-1][k][0][0], avg[i-1][k][0][1], avg[i-1][k][0][2]]),int(pbc),avg[i-1][k][1]]) # first frame coordinates
273 tracking_objects[track_ID].append([i,np.array([avg[i][j][0][0], avg[i][j][0][1], avg[i][j][0][2]]),int(pbc),avg[i][j][1]]) # second frame coordinates
274 track_ID += 1
275 else:
276 displacement = avg[i][j][0][0] + x_lim - avg[i-1][k][0][0]
277 if abs(displacement) < d_cutoff:
278 pbc = True
279 tracking_objects[track_ID].append([i-1,np.array([avg[i-1][k][0][0], avg[i-1][k][0][1], avg[i-1][k][0][2]]),int(pbc),avg[i-1][k][1]]) # first frame coordinates
280 tracking_objects[track_ID].append([i,np.array([avg[i][j][0][0], avg[i][j][0][1], avg[i][j][0][2]]),int(pbc),avg[i][j][1]]) # second frame coordinates
281 track_ID += 1
282
283 # For the remaining frames we compare the current x coordinate
284 # to those of the already existing objects
285 for i in range(2,len(avg)):
286 for j in range(len(avg[i])):
287 d_i = []
288 for obj_ID, obj in tracking_objects.items():
289 # Calculate the distances between the current avg x position
290 # and the last x position added to each tracking object
291 dist_1 = avg[i][j][0][0] - obj[-1][1][0]
292 dist_2 = avg[i][j][0][0] + x_lim - obj[-1][1][0]
293 dist_3 = avg[i][j][0][0] - x_lim - obj[-1][1][0]
294
295 # Determine whether dislocation has crossed periodic boundary
296 if abs(dist_2) < abs(dist_1) and abs(dist_2) < abs(dist_3):
297 dist = abs(dist_2)
298 pbc = 1
299 elif abs(dist_3) < abs(dist_1) and abs(dist_3) < abs(dist_2):
300 dist = abs(dist_3)
301 pbc = 3
302 else:
303 dist = abs(dist_1)
304 pbc = 0
305
306 d_i.append([avg[i][j][0][0], avg[i][j][0][1], avg[i][j][0][2], obj_ID, dist, pbc])
307
308 # Append coordinate to corresponding dislocation
309 # Find the minimum distance between the current average position
310 # and the average position in each dislocation in the previous frame
311 d_i = np.array(d_i)
312 min_dist = np.amin(d_i, axis=0)[4]
313 indx = np.where(d_i == min_dist)[0][0]
314 x = d_i[indx][0]
315 y = d_i[indx][1]
316 z = d_i[indx][2]
317 pbc = int(d_i[indx][5])
318
319 tracking_objects[indx].append([i, np.array([x,y,z]), pbc, avg[i][j][1]])
320
321 # Correct for PBCs
322 pbcs = []
323 for i, object in tracking_objects.items():
324 pbcs_i = []
325 for j in range(len(object)):
326 pbcs_i.append(object[j][2])
327 pbcs.append(np.array(pbcs_i))
328
329 starts = []
330 test = []
331 for pbc in pbcs:
332 starts.append(np.where(pbc == 1)[0])
333 test.append(np.where(pbc ==3)[0])
334
335
336 for i, start in enumerate(starts):
337 for s in start:
338 for j in range(s,len(pbcs[i])):
339 tracking_objects[i][j][1][0] += x_lim
340
341 for i, start in enumerate(test):
342 for s in start:
343 for j in range(s,len(pbcs[i])):
344 tracking_objects[i][j][1][0] -= x_lim
345
346 # Collect position data after tracking
347 position = []
348 vertices = []
349 t = []
350 for i, object in tracking_objects.items():
351 position_i = []
352 vertices_i = []
353 t_i = []
354 for j in range(len(object)):
355 position_i.append(math.hypot(object[j][1][0]))
356 vertices_i.append(object[j][3])
357 t_i.append(object[j][0]*0.001*500)
358 position.append(position_i)
359 vertices.append(vertices_i)
360 t.append(t_i)
361
362 # Change vertices from tuples to arrays for easier handling
363 # Sort vertices along y coords
364 disloc = []
365 for i in range(len(vertices)):
366 frames = []
367 for j in range(len(vertices[i])):
368 vertex = []
369 for k in range(len(vertices[i][j])):
370 # Wrap y
371 v_x = vertices[i][j][k][0]
372 v_y = vertices[i][j][k][1]
373 v_z = vertices[i][j][k][2]
374 vertex.append([v_x, v_y, v_z])
375 frames.append(sorted(vertex, key=lambda v: v[1]))
376 disloc.append(frames)
377
378 # Unwrap x to be consistent with average position data
379 for i, start in enumerate(starts):
380
381 for s in start:
382 for j in range(s,len(pbcs[i])):
383 for k in range(len(disloc[i][j])):
384 disloc[i][j][k][0] += x_lim
385
386 return position, disloc, t
