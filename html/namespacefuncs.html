61def get_posterior(data, n_samples, burn, dims=3):
62 """
63 Function to get parameter posterior distributions
64 Returns posterior samples for each parameter discarding burn samples
65 Also returns the multivariate posterior distribution
66 """
67 # All posterior samples
68 posterior = np.array(data["posterior"][burn:n_samples])
69 # Marginal posterior samples
70 c_dist = np.array([np.array(posterior[:,i]) for i in range(dims)])
71 # Parameter means
72 c_means = np.array([c_dist[i].mean() for i in range(dims)])
73 # Posterior covariance matrix
74 post_cov = np.array(np.cov(posterior, rowvar=False))
75 # Parameter standard deviations
76 c_std = np.array([np.sqrt(post_cov[i][i]) for i in range(dims)])
